# 0.3 — Glob Package Paths: Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend `packages:` in `neutron.yml` to accept full glob grammar — `**` recursive, `!` exclusion, `{a,b}` brace expansion — with empty-match warnings and a zero-packages-total error.

**Architecture:** Swap the current hand-rolled single-level `*` expansion in `src/graph/glob.ts` for `tinyglobby` (already a transitive dep via the TS plugin, version 0.2.15). Preserve directory-only filtering and order-preserving dedup. Emit a warning when a glob matches nothing; error at graph-build time when the whole workspace has zero packages.

**Tech Stack:** TypeScript, Bun test, `tinyglobby` (no new runtime dep), `zod` (unchanged — `packages: string[]` already matches the new grammar).

**Spec reference:** `specs/2026-04-18-v1-design.md` — Section *0.3 — Glob package paths*.

## Task 5: Replace glob implementation with tinyglobby

**Files:**
- Modify: `src/graph/glob.ts` (full rewrite; current file is 83 lines of manual regex)

- [ ] **Step 5.1: Rewrite the implementation**

Replace the entire contents of `src/graph/glob.ts` with:

```typescript
import { globSync } from "tinyglobby";

import { YELLOW, RESET } from "@/output";

/**
 * Expand `packages:` patterns against the workspace root.
 *
 * Accepts the full glob grammar (via tinyglobby):
 *   - literals:        `apps/web`
 *   - single-level:    `apps/*`
 *   - recursive:       `services/**`
 *   - brace expansion: `{apps,tools}/*`
 *   - exclusions:      `!packages/experimental-*` (filters matches from earlier patterns)
 *
 * Only directories are returned. Duplicates are removed while preserving the order
 * of first appearance (literal paths come first, then glob expansions alphabetically).
 *
 * Emits a warning to stderr when a single include-pattern matches zero packages
 * (not an error — users may be building toward packages that don't exist yet).
 */
export function expandPackageGlobs(
  patterns: readonly string[],
  root: string,
): readonly string[] {
  if (patterns.length === 0) {
    return [];
  }

  const includes: string[] = [];
  const excludes: string[] = [];
  for (const p of patterns) {
    if (p.startsWith("!")) {
      excludes.push(p.slice(1));
    } else {
      includes.push(p);
    }
  }

  const seen = new Set<string>();
  const results: string[] = [];

  for (const pattern of includes) {
    const matches = globSync(pattern, {
      cwd: root,
      onlyDirectories: true,
      ignore: excludes,
      dot: false,
      expandDirectories: false,
    });

    if (matches.length === 0) {
      console.error(
        `${YELLOW}warning${RESET}: packages pattern "${pattern}" matched no packages`,
      );
      continue;
    }

    // Normalize: tinyglobby may return trailing slashes on directories; strip them.
    // Alphabetize for predictable ordering across runs.
    const normalized = matches
      .map((m) => (m.endsWith("/") ? m.slice(0, -1) : m))
      .sort();

    for (const path of normalized) {
      if (!seen.has(path)) {
        seen.add(path);
        results.push(path);
      }
    }
  }

  return results;
}
```

function makeTempDir(): string {

const stubParser: ManifestParser = {

    return {
      name: "stub",
      version: "0.0.0",
      dependencies: [],
      localDependencyPaths: [],
    };

describe("buildWorkspaceGraph — package discovery", () => {
  test("errors when every ecosystem's glob expansion is empty", async () => {
    const root = makeTempDir();
    // No package directories created — every glob will match nothing.

    const config: NeutronConfig = {
      workspace: "demo",
      ecosystems: {
        typescript: {
          manifest: "package.json",
          packages: ["packages/*"],
        },
      },
      bridges: [],
    };

    const parsers = new Map([["package.json", stubParser]]);

    await expect(buildWorkspaceGraph(config, root, parsers)).rejects.toThrow(

    const config: NeutronConfig = {
      workspace: "demo",
      ecosystems: {
        typescript: {
          manifest: "package.json",
          packages: ["packages/*"],
        },
      },
      bridges: [],
    };

    const parsers = new Map([["package.json", stubParser]]);

    const graph = await buildWorkspaceGraph(config, root, parsers);
    expect(graph.packages.size).toBe(1);
  });
});
